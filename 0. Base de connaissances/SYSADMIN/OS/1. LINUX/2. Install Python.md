sudo make -j$(nproc)
sudo make altinstall

echo "Vérification de l'installation :"

# 3.8

```
#!/bin/bash

set -e

# Define the version
PYTHON_VERSION=3.8.18

echo "Installing dependencies..."
sudo apt update && sudo apt install -y \
  wget build-essential libssl-dev zlib1g-dev \
  libncurses5-dev libncursesw5-dev libreadline-dev \
  libsqlite3-dev libgdbm-dev libdb5.3-dev libbz2-dev \
  libexpat1-dev liblzma-dev tk-dev libffi-dev uuid-dev

echo "Downloading Python $PYTHON_VERSION..."
cd /usr/src
sudo wget https://www.python.org/ftp/python/$PYTHON_VERSION/Python-$PYTHON_VERSION.tgz --no-check-certificate

echo "Extracting source..."
sudo tar xzf Python-$PYTHON_VERSION.tgz
cd Python-$PYTHON_VERSION

echo "Building Python $PYTHON_VERSION..."
sudo ./configure --enable-optimizations
sudo make -j$(nproc)

echo "Installing Python $PYTHON_VERSION..."
sudo make altinstall

echo "Cleaning up..."
cd ..
sudo rm -rf Python-$PYTHON_VERSION*
echo "Done!"

# Check installation
echo "Python 3.8 installed at:"
which python3.8
python3.8 --version
```

# 3.10

```
#!/bin/bash
set -e

echo "### Mise à jour des paquets ###"
sudo apt update
sudo apt upgrade -y

echo "### Installation des dépendances nécessaires à la compilation de Python ###"
sudo apt install -y \
    build-essential \
    libssl-dev \
    zlib1g-dev \
    libbz2-dev \
    libreadline-dev \
    libsqlite3-dev \
    wget \
    curl \
    llvm \
    libncursesw5-dev \
    xz-utils \
    tk-dev \
    libxml2-dev \
    libxmlsec1-dev \
    libffi-dev \
    liblzma-dev \
    libgdbm-dev \
    libnss3-dev

# Choix du dossier temporaire pour installer python
TMP_DIR="/tmp/python3.10_install"

echo "### Création du dossier temporaire $TMP_DIR ###"
mkdir -p $TMP_DIR
cd $TMP_DIR

echo "### Téléchargement de Python 3.10.13 (version stable) ###"
wget https://www.python.org/ftp/python/3.10.13/Python-3.10.13.tgz

echo "### Extraction de l'archive ###"
tar -xf Python-3.10.13.tgz
cd Python-3.10.13

echo "### Configuration et compilation de Python 3.10 ###"
./configure --enable-optimizations
make -j$(nproc)

echo "### Installation (nécessite sudo) ###"
sudo make altinstall

echo "### Nettoyage ###"
cd ~
rm -rf $TMP_DIR

echo "### Installation terminée. Vérification de la version ###"
python3.10 --version

echo "### Optionnel : mise à jour de python3 vers python3.10 ###"
echo "Pour mettre python3 vers python3.10, fais :"
echo "sudo update-alternatives --install /usr/bin/python3 python3 /usr/local/bin/python3.10 1"
echo "Puis pour choisir la version :"
echo "sudo update-alternatives --config python3"

```
